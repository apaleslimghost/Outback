  \prop{Cost}{20 scrap}
  Melee weapon, with maximum damage of 12. Can also be used to pry open containers, locked doors etc.. You make a \stat{Brawn} roll against the object's \stat{Ruggedness}.
\end{describe}

\begin{describe}{Sniper rifle}
  \prop{Cost}{140 scrap}
  \prop{Rounds}{15 scrap each}
  Ranged weapon, with range 500 feet and maximum damage of 30. The attacking skill is \stat{Brains (Perception)} and \stat{Brains (Ranged weapons)} combined.

  \prop{Cost}{80 scrap}
  \prop{Rounds}{10 scrap each}

\begin{describe}{9mm pistol}
  \prop{Cost}{50 scrap}
  \prop{Rounds}{5 scrap each}

\begin{describe}{Blackthumb's tools}
  \prop{Cost}{100 scrap}

\section{Actions}
\newcommand{\duration}[1]{
\begin{switch}{#1}%
  \case{1}{Instant (1 tick)}%
  \case{2}{Quick (2 ticks)}%
  \case{3}{Steady (3 ticks)}%
  \case{5}{Slow (5 ticks)}%
  \case{8}{Long (8 ticks)}%
\end{switch}}

\begin{describe}{Repair}
  \prop{Cooldown}{\duration{8}}
  Choose a vehicle you have access to or character within 3 feet. Roll \die{2d8} plus half your \stat{Brains (Repair)} modifier. Add the result to the HP of the target.
\end{describe}

\begin{describe}{Quick repair}
  \prop{Cooldown}{\duration{2}}
  Choose a vehicle you have access to or character within 3 feet. Roll \stat{Brains (Repair)} against the target's damage (their \stat{Max HP} minus their \stat{Current HP}). If you succeed, the damage is removed.
\end{describe}

\begin{describe}{Remove Affliction}
  \prop{Cooldown}{\duration{5}}
  Choose a vehicle you have access to or character within 3 feet. Roll \stat{Brains (Repair)} against the target's damage (their \stat{Max HP} minus their \stat{Current HP}). If you succeed, the damage is removed.
\end{describe}

\section{Man\oe{}uvres}

\begin{describe}{Accelerate}
  \prop{Fuel}{2}
  \prop{Cooldown}{\duration{1}}
  Roll \die{2d8} plus half your \stat{Wit (Driving) + Acceleration} modifier. Add the result to your \stat{Speed}.
\end{describe}

\begin{describe}{Maintain speed}
  \prop{Fuel}{0}
  \prop{Cooldown}{\duration{1}}
  Your \stat{Speed} does not change at the end of the tick.
\end{describe}

\begin{describe}{Brake}
  \prop{Fuel}{0}
  \prop{Cooldown}{\duration{2}}
  Roll \die{2d8} plus half your \stat{Wit (Driving) + Braking} modifier. Subtract the result from your \stat{Speed}.
\end{describe}

\begin{describe}{Handbrake}
  \prop{Fuel}{0}
  \prop{Cooldown}{\duration{8}}
  At the end of the tick, your Speed drops to 0.
\end{describe}

\begin{describe}{Corner}
  \prop{Fuel}{0}
  \prop{Cooldown}{\duration{1}}
  Roll \stat{Wit (Driving) + Handling}
\end{describe}

\section{Afflictions}

\begin{describe}{Blowout}
  Subtract 5 from your vehicle's \stat{Handling} and 50 from its \stat{Top Speed}.
\end{describe}

\begin{describe}{Engine fire}
  Subtract 5 from your vehicle's \stat{Acceleration}. If this affliction is not removed in 5 ticks (including the tick it was added), your vehicle explodes, killing any occupants.
\end{describe}
