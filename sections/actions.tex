\begin{describe}{Hand-to-hand}
  \prop{Cooldown}{\duration{2}}
  \prop{Range}{melee}
  \prop{Maximum damage}{5}
  \stat[Hand-to-hand]{Brawn} vs \stat[Armour]{Brawn} \orr \stat[Agility]{Wit}
\end{describe}

\begin{describe}{Melee weapon}
  \stat[Melee weapons]{Brawn} vs \stat[Armour]{Brawn} \orr \stat[Agility]{Wit}
\end{describe}

\begin{describe}{Ranged weapon}
  \stat[Ranged weapons]{Brains} vs \stat[Armour]{Brawn} \orr \stat[Agility]{Wit}
\end{describe}

\begin{describe}{Shunt}
  \prop{Cooldown}{\duration{5}}
  \prop{Range}{vehicle within 25 feet with speed lower than yours}
  \prop{Maximum damage}{20}
  \combined{\stat[Driving]{Wit}}{\stat{Speed}} vs \combined{\stat{Speed}}{\stat{Ruggedness}}. At the end of the tick, your speed becomes the same as the target's. You may not man\oe{}uvre in this tick.
\end{describe}

\begin{describe}{Ram}
  \prop{Cooldown}{\duration{8}}
  \prop{Range}{vehicle within 25 feet that is stationary or travelling in the opposite direction to you}
  \prop{Maximum damage}{30}
  \combined{\stat[Driving]{Wit}}{\stat{Speed}} vs \stat{Ruggedness}. At the end of the tick, your and the target's speed becomes zero. Neither of you may man\oe{}uvre in this tick.
\end{describe}

\begin{describe}{Sideswipe}
  \prop{Cooldown}{\duration{3}}
  \prop{Range}{vehicle within 25 feet}
  \prop{Maximum damage}{10}
  \combined{\stat[Driving]{Wit}}{\stat{Handling}} vs \combined{\stat[Driving]{Wit}}{\stat{Ruggedness}}
\end{describe}

\begin{abstractsection}{Repair actions}
\vspace{1ex}

\begin{describe}{Repair}
  \prop{Cooldown}{\duration{8}}
  Choose a vehicle you have access to or character within 3 feet. Roll \die{1d8} plus  your \stat[Repair]{Brains} modifier. Add the result to the HP of the target.
\end{describe}

\begin{describe}{Quick repair}
  \prop{Cooldown}{\duration{2}}
  Choose a vehicle you have access to or character within 3 feet. The target's HP increases at a cost of 6 scrap minus your \stat[Repair]{Brains} modifier per hit point.
\end{describe}

\begin{describe}{Remove Affliction}
  \prop{Cooldown}{\duration{8}}
  Requires Blackthumb's Tools. Choose a vehicle you have access to or character within 3 feet. Roll \stat[Repair]{Brains} against the difficulty of an affliciton of the target.  If you succeed, the affliction is removed.
\end{describe}
\end{abstractsection}
