\begin{describe}{Blowout}
  \prop{Difficulty}{+1}
  Subtract 5 from the vehicle's \stat{Handling} and 50 from its \stat{Top Speed}. If your \stat{Speed} is above your temporary \stat{Top Speed}, it immediately reduces to the \stat{Top Speed}.
\end{describe}

\begin{describe}{Engine fire}
  \prop{Difficulty}{+4}
  Subtract 5 from the vehicle's \stat{Acceleration}. If the affliction isn't removed in 5 ticks, the vehicle explodes, killing any occupants.
\end{describe}
