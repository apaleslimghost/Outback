\begin{describe}{Blowout}
  Subtract 5 from your vehicle's \stat{Handling} and 50 from its \stat{Top Speed}.
\end{describe}

\begin{describe}{Engine fire}
  Subtract 5 from your vehicle's \stat{Acceleration}. If this affliction is not removed in 5 ticks (including the tick it was added), your vehicle explodes, killing any occupants.
\end{describe}
